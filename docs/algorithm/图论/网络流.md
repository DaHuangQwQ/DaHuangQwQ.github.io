## **EK**增广路算法

> ***O\*(\*n\**m\*2)**

```cpp
const int N = 5e5 + 10;
// https://www.luogu.com.cn/problem/P3376

int n,m,s,t;
int flag[2510][2510];
//
struct Edge {
    int to, next;
    ll w;
} e[N << 1];
int head[N];
static int idx = 1; // [!code warning]
void add(int u, int v, int w){
    e[++idx] = (Edge) {v, head[u], w};
    head[u] = idx;
}
//
int mf[N], pre[N];
bool bfs(){ // 找 增广路
    memset(mf, 0, sizeof mf);
    queue<int> q;
    q.push(s);
    mf[s] = 0x3f3f3f3f3f3f3f3f;
    while(!q.empty()){
        int u = q.front(); // [!code warning]
        q.pop();
        for(int i = head[u];i;i = e[i].next){
            int v = e[i].to;
            int w = e[i].w;
            // cout << v << endl;
            if(mf[v] == 0 && w){
                mf[v] = min(mf[u], w);
                pre[v] = i;
                q.push(v);
                if(v == t) return true;
            }
        }
    }
    return false;
}

ll EK(){
    ll flow = 0;
    while(bfs()){
        int v = t;
        while(v != s){
            int i = pre[v];
            e[i].w -= mf[t];
            e[i^1].w += mf[t];
            v = e[i^1].to;
        }
        debug(">%d\n", mf[t]);
        flow += mf[t];
    }
    return flow;
}


void solve(){
    cin >> n >> m >> s >> t;
    for(int i = 1;i <= m;i++){
        int u,v,w; cin >> u >> v >> w;
        if(flag[u][v]==0) {  //处理重边的操作（加上这个模板题就可以用Ek算法过了） 
			add(u,v,w);
            add(v,u,0);
			flag[u][v]=idx;
		}
		else {
			e[flag[u][v]-1].w+=w;
		}
    }
    cout << EK() << endl;;
}

signed main(){

    #endif

    cin.tie(0);cout.tie(0);
    int n = 1;
    // cin >> n;

        solve();
    }
    return 0;
}
```

## Dinic

1. 搜索顺序优化（分层限制搜索深度）

2. 当前弧优化（剪枝）

3. 剩余流量优化（剪枝）

4. 残枝优化（剪枝）

   

   >  分层图&DFS

```cpp
const int N = 1e5 + 10;
// https://www.luogu.com.cn/problem/P2740
int n, m, s, t;
// 
struct Edge {
    int to, next;
    int w;
} e[N << 1];
int head[N];
static int idx = 1;
void add(int u, int v, int w){

    e[++idx] = (Edge) {v, head[u], w};
    head[u] = idx;
}
//
int d[N], cur[N];
bool bfs(){
    memset(d, 0, sizeof d);
    queue<int> q;
    q.push(s);
    d[s] = 1;
    // for(int i = 1;i <= n;i++) cur[i]=head[i];
    while(!q.empty()){
        int u = q.front();
        q.pop();
        for(int i = head[u];i;i = e[i].next){
            int v = e[i].to;
            int w = e[i].w;
            if(d[v] == 0 && w){
                debug(">%d %d\n", u, v);
                d[v] = d[u] + 1;
                q.push(v);
                if(v == t) return true;
            }
        }
    }
    return false;
}
ll dfs(int u, int mf){
    if(u == t) return mf;
    ll sum = 0;
    for(int i = head[u];i;i = e[i].next){
        cur[u] = i; // 当前弧 优化
        int v = e[i].to;
        int w = e[i].w;
        if(d[v] == d[u] + 1 && w){
            ll f = dfs(v, min(mf, w));
            e[i].w -= f;
            e[i^1].w += f;
            sum += f;
            mf -= f; // 剩余流量
            if(mf == 0) break; // 余量优化
        }
    }
    if(sum == 0) d[u] = 0; // 残枝优化
    return sum;
}

ll dinic(){
    ll flow = 0;
    while(bfs()){
        memcpy(cur, head, sizeof head);
        flow += dfs(s, 1e9);
    }
    return flow;
}

int flag[1000][1000];
void solve(){
    cin >> m >> n;
    s = 1;
    t = n;
    for(int i = 1;i <= m;i++){
        int u, v, w; cin >> u >> v >> w;
        add(u,v,w);
        add(v,u,0);
    }
    cout << dinic() << endl;
}
```

## 最小割

> 最小割就是求得一个割（S, T），使得割的容量c（S, T）最小。
>
> 最小割的方案往往并不是唯一的。



> 最大流 = 最小割

1. 求最小割
2. 求最小割的划分
3. 求最小割的最少边数

## 最小费用最大流

```cpp
const int N = 1e5 + 10;
// https://www.luogu.com.cn/problem/P3381
int n, m, s, t;
//
struct Edge {
    int to, next;
    ll w, c;
} e[N << 1];
int head[N];
void add(int u,int v,int w,int c){
    static int idx = 1;
    e[++idx] = (Edge) {v, head[u], w, c};
    head[u] = idx;
}
//
int mf[N], pre[N];
int dis[N];
bool vis[N];
bool spfa(){
    memset(dis, 0x3f, sizeof dis);
    memset(mf, 0, sizeof mf);
    
    queue<int> q;
    q.push(s);
    dis[s] = 0;
    mf[s] = 1e8;
    vis[s] = true;
    while(!q.empty()){
        int u = q.front();
        q.pop();
        vis[u] = false;
        for(int i = head[u];i;i = e[i].next){
            int v = e[i].to;
            int w = e[i].w;
            int c = e[i].c;
            if(dis[v] > dis[u] + w && c){
                debug(">%d %d\n", u, v);
                dis[v] = dis[u] + w;
                mf[v] = min(mf[u], c);
                pre[v] = i;
                if(!vis[v]){
                    q.push(v);
                    vis[v] = true;
                }
            }
        }
    }
    return mf[t] > 0;
}
//
ll flow = 0;
ll cost = 0;
void EK(){
    while(spfa()){
        for(int v = t;v != s;){
            int i = pre[v];
            e[i].c -= mf[t];
            e[i^1].c += mf[t];
            v = e[i^1].to;
        }
        flow += mf[t];
        cost += mf[t] * dis[t];
    }
}

void solve(){
    cin >> n >> m >> s >> t;
    for(int i = 1;i <= m;i++){
        int u,v,w,c; cin >> u >> v >> c >> w;
        add(u,v,w,c);
        add(v,u,-w,0);
    }
    EK();
    cout << flow << " " << cost << endl;
}
```
