struct matrix{
    ll c[32][32];
    matrix(){
        memset(c, 0, sizeof(c));
    }
} A, B , res;

ll MOD = 1e9 + 7;
matrix operator*(matrix & x, matrix & y){
    matrix t;
    for(int i = 1; i <= n;i++){
        for(int j = 1;j <= n;j++){
            for(int k = 1; k <= n;k++){
                A.c[i][j] = (A.c[i][j] + x.c[i][k] * y.c[k][j]) % MOD;
            }
        }
    }
    return t;
}

}
