const int N = 1e3 + 10;

int n,m;
int board[N][N];
struct node{
    int x, y, w;
};
bool vis[N][N];

int dx[] = {1,-1,0,0};
int dy[] = {0,0,1,-1};

int bfs(int x, int y){
    memset(vis, false, sizeof vis);

    int res = 1;
    queue<node> q;
    q.push((node){x,y,1});
    vis[x][y] = true;
    while(!q.empty()){
        node t = q.back();
        q.pop();
        for(int i = 0; i < 4;i++){
            int x = dx[i] + t.x;
            int y = dy[i] + t.y;
            int w = t.w;
            if(x < 1 || x > n || y < 1 || y > n || vis[x][y] == true) continue;
            if(board[x][y] == board[t.x][t.y]) continue;
            // cout << x << " " << y << " " << w << endl;
            q.push((node){x,y,w+1});
            vis[x][y] = true;
            res++;
        }
    }
    return res;
}
